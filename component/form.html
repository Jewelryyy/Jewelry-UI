Form

由输入框、选择器、单选框、多选框等控件组成，用以收集、校验、提交数据。


典型表单

包括各种表单项，比如输入框、选择器、开关、单选框、多选框等。

<form class="form">
    <!-- 输入框 -->
    <div class="input">
        <label>
            输入框：
            <input type="text" name="input" placeholder="请输入文本">
        </label>
    </div>
    <br />
    <!-- 选择器 -->
    <div class="select">
        <label>
            选择器：
            <select name="options">

            </select>
        </label>
    </div>
    <br />
    <!-- 计数器 -->

        <input type="number" id="count" value="0">
    </div>
    <br />
    <!-- 开关 -->
    <div class="switch_div">
        <label>开关：</label>
        <label class="switch">
            <input type="checkbox">
            <span class="slider"></span>
        </label>
    </div>
    <br />
    <!-- 单选框 -->
    <div class="radio">
        <label>单选框：</label>
        <label>
            <input type="radio" name="option" value="1">
            <span class="custom-radio"></span>
            选项1
        </label>
        <label>
            <input type="radio" name="option" value="2">
            <span class="custom-radio"></span>
            选项2
        </label>
    </div>
    <br />
    <!-- 多选框 -->
    <div class="checkbox">
        <label>多选框：</label>
        <label>
            <input type="checkbox" name="option" value="1">
            <span class="custom-checkbox"></span>
            选项1
        </label>
        <label>
            <input type="checkbox" name="option" value="2">
            <span class="custom-checkbox"></span>
            选项2
        </label>
        <label>

            <span class="custom-checkbox"></span>
            选项3
        </label>
    </div>
    <br />
    <!-- 颜色选择器 -->
    <div class="colorPicker">
        <label>
            选择颜色：
            <input type="color" name="color">
        </label>
    </div>
    <br />
    <!-- 提交按钮 -->
    <button class="btn primary">提交</button>
    <!-- 重置按钮 -->
    <button class="btn secondary">重置</button>
    <br />
</form>
